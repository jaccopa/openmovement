00055 {
00056     FILE *ofp = NULL;

00059 
00060     /* Open the binary file reader on the input file */
00061     reader = OmReaderOpen(infile);
00062     if (reader == NULL)
00063     {
00064         printf("ERROR: Problem opening file: %s\n", infile);
00065         return -1;
00066     }
00067 
00068     /* Open the output file (or stdout if none) */
00069     if (outfile != NULL)
00070     {
00071         ofp = fopen(outfile, "wt");
00072     }
00073     if (ofp == NULL)
00074     {
00075         ofp = stdout;
00076     }
00077 
00078 //#define VERIFY
00079 #ifdef VERIFY
00080     OmReaderDataBlockSeek(reader, 82875);
00081 #endif
00082 
00083     /* Iterate over all of the blocks of the file */
00084     while ((numSamples = OmReaderNextBlock(reader)) >= 0)
00085     {
00086         short *buffer;
00087         int i;
00088 
00089         buffer = OmReaderBuffer(reader);
00090         for (i = 0; i < numSamples; i++)
00091         {
00092             OM_DATETIME dateTime;
00093             unsigned short fractional;
00094             short x, y, z;
00095             static char timeString[25];  /* "YYYY-MM-DD hh:mm:ss.000"; */
00096 #ifdef VERIFY
00097             static int firstPacket = 1, seconds, lastSecond = -1, packetCount = 0;
00098 #endif
00099 
00100             /* Get the date/time value for this sample, and the 1/65536th fractions of a second */
00101             dateTime = OmReaderTimestamp(reader, i, &fractional);
00102             sprintf(timeString, "%04d-%02d-%02d %02d:%02d:%02d.%03d", 
00103                     OM_DATETIME_YEAR(dateTime), OM_DATETIME_MONTH(dateTime), OM_DATETIME_DAY(dateTime), 
00104                     OM_DATETIME_HOURS(dateTime), OM_DATETIME_MINUTES(dateTime), OM_DATETIME_SECONDS(dateTime), 
00105                     (int)fractional * 1000 / 0xffff);
00106 
00107             /* Get the x/y/z/ values */
00108             x = buffer[3 * i + 0];
00109             y = buffer[3 * i + 1];
00110             z = buffer[3 * i + 2];
00111 
00112             /* Output the data */
00113             fprintf(ofp, "%s,%d,%d,%d\n", timeString, x, y, z);
00114 
00115 #ifdef VERIFY
00116 seconds = OM_DATETIME_SECONDS(dateTime);
00117 if (lastSecond == -1) { lastSecond = seconds; };
00118 if (seconds != lastSecond)
00119 {
00120     if (firstPacket) { printf(","); firstPacket = 0; }
00121     else if (packetCount >= 98 && packetCount <= 105) { printf("."); }
00122     else
00123     { 
00124         if (seconds == lastSecond + 1 || (seconds == 0 && lastSecond == 59))
00125         {
00126             fprintf(stderr, "\r\nWARNING: Only %d samples in second :%02d before %s]", packetCount, lastSecond, timeString);
00127         }
00128         else
00129         {
00130             fprintf(stderr, "\r\nWARNING: Non-sequential second jump (%d samples in second :%02d before %s)]", packetCount, lastSecond, timeString);
00131         }
00132         printf("\r\n");
00133     }
00134     lastSecond = seconds;
00135     packetCount = 0;
00136 }
00137 packetCount++;
00138 #endif
00139 
00140             /* 'Tee' the data to stderr */
00141             if (tee) { fprintf(stderr, "%s,%d,%d,%d\n", timeString, x, y, z); }
00142         }
00143     }
00144 
00145     /* Close the files */
00146     OmReaderClose(reader);
00147     if (ofp != NULL && ofp != stdout) { fclose(ofp); }
00148     return 0;
