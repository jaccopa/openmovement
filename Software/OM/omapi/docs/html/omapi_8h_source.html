00762                 ( (((OM_DATETIME)((year) % 100) & 0x3f) << 26) \
00763                 | (((OM_DATETIME)(month)        & 0x0f) << 22) \
00764                 | (((OM_DATETIME)(day)          & 0x1f) << 17) \
00765                 | (((OM_DATETIME)(hours)        & 0x1f) << 12) \
00766                 | (((OM_DATETIME)(minutes)      & 0x3f) <<  6) \
00767                 | (((OM_DATETIME)(seconds)      & 0x3f)      ) \
00768                 )
00769 #define OM_DATETIME_YEAR(dateTime)    ((unsigned int)((unsigned char)(((dateTime) >> 26) & 0x3f)) + 2000) 
00770 #define OM_DATETIME_MONTH(dateTime)   ((unsigned char)(((dateTime) >> 22) & 0x0f))  
00771 #define OM_DATETIME_DAY(dateTime)     ((unsigned char)(((dateTime) >> 17) & 0x1f))  
00772 #define OM_DATETIME_HOURS(dateTime)   ((unsigned char)(((dateTime) >> 12) & 0x1f))  
00773 #define OM_DATETIME_MINUTES(dateTime) ((unsigned char)(((dateTime) >>  6) & 0x3f))  
00774 #define OM_DATETIME_SECONDS(dateTime) ((unsigned char)(((dateTime)      ) & 0x3f))
